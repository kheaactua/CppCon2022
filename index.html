#include <array>
#include <algorithm>
#include <type_traits>
#include <variant>

template <class...> struct type_list {};

template <size_t I, class... Ts>
using at_t = std::variant_alternative_t<I+1, std::variant<std::monostate, Ts...>>;

template <class List, auto Key> struct sort;
template <class... Ts, auto Key>
struct sort<type_list<Ts...>, Key> : std::type_identity<decltype(
  []<size_t... Idxs>(std::index_sequence<Idxs...>){
    constexpr auto sorted_indices = []{
      std::array idxs = {Idxs...};
      std::ranges::sort(idxs, [](size_t idx1, size_t idx2) {
        using variant_t = std::variant<std::type_identity<Ts>...>;
        std::array vars = {
          variant_t{ std::in_place_index_t<Idxs>{} }...
        };
        return std::visit(Key, vars[idx1]) < std::visit(Key, vars[idx2]);
      });
      return idxs;
    }();
    return type_list<at_t<sorted_indices[Idxs], Ts...>...>{};
  }(std::index_sequence_for<Ts...>{})
)> {};

static_assert(std::is_same_v<
  sort<
    type_list<int[3], int[1], int[4]>,
    /* Key = */ []<class T>(std::type_identity<T>) {
      return sizeof(T);  // for instance
    }
  >::type,
  type_list<int[1], int[3], int[4]>
>);

Type Erasure

            
    
              
  • hide the exact type of an object when you go to work with it
    • 
              
  • can work with any type that might be declared in the future
    • 
              
  • compilation firewall to prevent recompiling the entire library for adding a new type
